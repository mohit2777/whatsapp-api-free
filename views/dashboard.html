API Documentation

Complete guide to integrate with WhatsApp Multi-Automation API

Quick Start

All API requests require authentication using an API key. Get your API key from the account details in the dashboard.

Headers 
Content-Type: application/json
X-API-Key: wak_your_api_key_here

Send Text Message

POST /api/send
cURL 
curl -X POST "https://your-domain.com/api/send" \
  -H "Content-Type: application/json" \
  -H "X-API-Key: wak_your_api_key" \
  -d '{
    "number": "919876543210",
    "message": "Hello from WhatsApp API!"
  }'
JavaScript (fetch) 
const response = await fetch('https://your-domain.com/api/send', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-API-Key': 'wak_your_api_key'
  },
  body: JSON.stringify({
    number: '919876543210',
    message: 'Hello from WhatsApp API!'
  })
});

const result = await response.json();
console.log(result);
Python 
import requests

response = requests.post(
    'https://your-domain.com/api/send',

        'number': '919876543210',

Send Media (Base64)

Send images, videos, documents, or audio files using Base64 encoded data.

POST /api/send-media

Image Example

cURL 
curl -X POST "https://your-domain.com/api/send-media" \
  -H "Content-Type: application/json" \
  -H "X-API-Key: wak_your_api_key" \
  -d '{
    "number": "919876543210",
    "caption": "Check out this image!",
    "media": {
      "data": "/9j/4AAQSkZJRgABAQEASABIAAD...",
      "mimetype": "image/jpeg",
      "filename": "photo.jpg"
    }
  }'

JavaScript - File to Base64

JavaScript (Browser) 
// Convert file input to Base64 and send
async function sendMediaFromFile(file) {
  // Read file as Base64
  const base64 = await new Promise((resolve) => {
    const reader = new FileReader();
    reader.onload = () => {
      // Remove "data:image/jpeg;base64," prefix
      const base64Data = reader.result.split(',')[1];
      resolve(base64Data);
    };
    reader.readAsDataURL(file);
  });

  const response = await fetch('https://your-domain.com/api/send-media', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'wak_your_api_key'
    },
    body: JSON.stringify({
      number: '919876543210',
      caption: 'Sent from browser!',
      media: {
        data: base64,
        mimetype: file.type,
        filename: file.name
      }
    })
  });

  return response.json();
}

// Usage with file input
const fileInput = document.querySelector('input[type="file"]');
fileInput.addEventListener('change', (e) => {
  sendMediaFromFile(e.target.files[0]);
});

Node.js - File to Base64

Node.js 
const fs = require('fs');
const path = require('path');

async function sendMediaFile(filePath, number, caption) {
  // Read file and convert to Base64
  const fileBuffer = fs.readFileSync(filePath);
  const base64Data = fileBuffer.toString('base64');
  
  // Determine mimetype from extension
  const ext = path.extname(filePath).toLowerCase();
  const mimeTypes = {
    '.jpg': 'image/jpeg',
    '.jpeg': 'image/jpeg',
    '.png': 'image/png',
    '.gif': 'image/gif',
    '.mp4': 'video/mp4',
    '.mp3': 'audio/mpeg',
    '.ogg': 'audio/ogg',
    '.pdf': 'application/pdf',
    '.doc': 'application/msword',
    '.docx': 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
  };
  
  const response = await fetch('https://your-domain.com/api/send-media', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'wak_your_api_key'
    },
    body: JSON.stringify({
      number: number,
      caption: caption,
      media: {
        data: base64Data,
        mimetype: mimeTypes[ext] || 'application/octet-stream',
        filename: path.basename(filePath)
      }
    })
  });

  return response.json();
}

// Usage
sendMediaFile('./invoice.pdf', '919876543210', 'Here is your invoice')
  .then(console.log);

Python - File to Base64

Python 
import base64
import mimetypes
import requests
from pathlib import Path

def send_media_file(file_path, number, caption, api_key):
    # Read and encode file
    with open(file_path, 'rb') as f:
        base64_data = base64.b64encode(f.read()).decode('utf-8')
    
    # Get mimetype
    mimetype, _ = mimetypes.guess_type(file_path)
    filename = Path(file_path).name
    
    response = requests.post(
        'https://your-domain.com/api/send-media',
        headers={
            'Content-Type': 'application/json',
            'X-API-Key': api_key
        },
        json={
            'number': number,
            'caption': caption,
            'media': {
                'data': base64_data,
                'mimetype': mimetype or 'application/octet-stream',
                'filename': filename
            }
        }
    )
    
    return response.json()

# Usage
result = send_media_file(
    './photo.jpg',
    '919876543210',
    'Check this out!',
    'wak_your_api_key'
)
print(result)

Send Media (URL)

Send media directly from a public URL. The server will download and forward it.

cURL 
curl -X POST "https://your-domain.com/api/send-media" \
  -H "Content-Type: application/json" \
  -H "X-API-Key: wak_your_api_key" \
  -d '{
    "number": "919876543210",
    "caption": "Image from URL",
    "media": {
      "url": "https://example.com/image.jpg"
    }
  }'

Delivery Receipt Payload

JSON 
{
  "event": "message_ack",
  "account_id": "uuid-of-account",
  "message_id": "3EB0ABC123...",
  "recipient": "919876543210@s.whatsapp.net",
  "ack": 3,
  "ack_name": "delivered",
  "timestamp": 1705747200
}

// ack values:
// 1 = pending (clock icon)
// 2 = sent (single tick)
// 3 = delivered (double tick)
// 4 = read (blue ticks)
